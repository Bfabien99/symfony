"cd" pour se deplacer
"rm" pour supprimer des fichiers
    -"rf": "r" recursif, "f" force

# SYMFONY
"symfony server:start" pour lancer le server
    -"d" lance le server en arrière plan

"symfony server:stop" pour stopper le server
"symfony open:local" pour ouvrir le navigateur à l'adresse du server

"symfony console make:controller controller_name" pour creer le controller
"symfony console debug:router" pour voir toutes ses routes

"symfony console doctrine:database:create" create database
"symfony console doctrine:fixtures:load" purge database

"symfony console make:form Entity_nameFormType Entity_name" create form
"composer require symfony/security-bundle" package for authentification
"symfony console make:user User" create user
"symfony console make:registration-form" create registration form
"symfony console make:auth" create login form
